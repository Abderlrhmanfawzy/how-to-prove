Solution (a)

Domain = $\{ p \mid \text{p is a living parent} \}$

\section{Problem 2}

Solution(a)

Domain = $\{ p \mid \text{p is a brother to some living person }\}$
Range = $\{q \mid \text{q has some living brother.}\}$

Solution(b)

Domain = $\mathbb{R}$
Range = $\{ x \mid x \in \mathbb{R} \land  x > -1 \land x < 1 \}$
